package algo0824;

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.LinkedList;
import java.util.Queue;

public class BOJ_16954_ì›€ì§ì´ëŠ”ë¯¸ë¡œíƒˆì¶œ {

	static int N = 8;
	static char[][] map;
	static int[] dx = {0, 1, 0, -1, 0, 1, -1, -1, 1};
	static int[] dy = {0, 0, 1, 0, -1, 1, 1, -1, -1};

	static class Pos {
		int x, y;

		public Pos(int x, int y) {
			this.x = x;
			this.y = y;
		}
	}

	public static void main(String[] args) throws Exception {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		map = new char[N][N];

		for (int i = 0; i < N; i++) {
			map[i] = br.readLine().toCharArray();
		}

		System.out.println(bfs());
	}

	static int bfs() {
		Queue<Pos> Q = new LinkedList<>();
		Q.add(new Pos(N - 1, 0));

		while (!Q.isEmpty()) {
			int cnt = Q.size();

			while(cnt-- > 0) {
				Pos cur = Q.poll();

				// ìºë¦­í„°ê°€ ìˆëŠ” ê³³ì— ë²½ì´ ì´ë™í–ˆë‹¤ë©´
				if (map[cur.x][cur.y] == '#') continue;

				// ìºë¦­í„° ì´ë™
				for (int dir = 0; dir < 9; dir++) {
					int nx = cur.x + dx[dir];
					int ny = cur.y + dy[dir];

					if (!isAvailable(nx, ny)) continue;
					if (nx == 0 && ny == N - 1) return 1;
					if (map[nx][ny] == '#') continue;

					Q.add(new Pos(nx, ny));
				}
			}

			// ë²½ ì´ë™
			moveWall();
		}

		return 0;
	}

	static void moveWall() {
		for (int i = N - 2; i >= 0; i--) {
			for (int j = 0; j < N; j++) {
				map[i + 1][j] = map[i][j];
			}
		}
		Arrays.fill(map[0], '.');
	}

	static boolean isAvailable(int x, int y) {
		return x >= 0 && y >= 0 && x < N && y < N;
	}
}
